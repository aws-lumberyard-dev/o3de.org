template<class Key, class MappedType, AZStd::size_t NumLocks = 8, class Hasher = AZStd::hash<Key>, class EqualKey = AZStd::equal_to<Key>, class Allocator = AZStd::allocator>
class AZStd::concurrent_unordered_map< Key, MappedType, NumLocks, Hasher, EqualKey, Allocator >

Concurrent unordered map container, similar to unordered_map, but uses a striped array of locks to allow concurrent access.
